    传统服务器运维常用命令手册¶

    文档类型: 工具手册 | 更新时间: 2026-06-04 | 维护人: 研发团队 | 标签: 服务器运维, Linux命令, Jar管理, 日志排查, 定时任务

    JAR 包管理¶

    查看 JAR 包信息¶

    # 查看 JAR 包大小和位置
ls -lh *.jar

# 查看 JAR 包内容
jar tf <jar-file>.jar | head -20

# 解压 JAR 包查看配置
unzip -l <jar-file>.jar | grep application

    进程管理¶

    查看进程¶

    # 查找 Java 进程
ps aux | grep java

# 查看特定服务的进程
ps aux | grep <服务名> | grep -v grep

# 查看完整启动命令
ps -ef | grep java | grep <服务名>

# 查看进程的启动参数
cat /proc/<pid>/cmdline | tr '\0' ' '

    端口管理¶

    # 查看所有 Java 进程监听的端口
netstat -tulnp | grep java
# 或
ss -tulnp | grep java

# 查看特定端口是否被占用
netstat -tulnp | grep <端口号>

# 查看指定服务的端口
netstat -tulnp | grep <服务名>

    停止/启动服务¶

    # 停止服务（优雅关闭）
kill -15 <pid>

# 强制停止服务
kill -9 <pid>

# 或者找到进程 ID 后停止
pkill -f "<服务关键词>"

# 确认进程已停止
ps aux | grep <服务名> | grep -v grep

    服务启动¶

    标准启动命令¶

    # 启动服务（传统方式）
nohup java -jar -Xms512m -Xmx2g <jar-file>.jar --spring.profiles.active=test > app.log 2>&1 &

# 启动服务（推荐方式 - 以 huayang 环境为例）
nohup java -server -Xms1024m -Xmx2048m \
  -XX:+HeapDumpOnOutOfMemoryError \
  -XX:HeapDumpPath=./heapdump \
  -XX:ErrorFile=./hs_err_pid%p.log \
  -Dfile.encoding=UTF-8 \
  -Duser.timezone=Asia/Shanghai \
  -Dspring.profiles.active=huayang \
  -jar <服务名>.jar \
  > <服务名>.log 2>&1 &

# 查看启动日志
tail -f <服务名>.log

# 检查启动是否成功
grep "Started.*Application" <服务名>.log

# 检查端口监听
netstat -tlnp | grep <端口号>

    完整重启流程¶

    # 1. 查找进程
ps aux | grep <服务名> | grep -v grep

# 2. 停止进程
kill -9 <PID>

# 3. 确认进程已停止
ps aux | grep <服务名> | grep -v grep

# 4. 清空旧日志（可选）
> <服务名>.log

# 5. 启动新服务
nohup java -server -Xms1024m -Xmx2048m -XX:+HeapDumpOn -XX:HeapDumpPath=./heapdump -XX:ErrorFile=./hs_err_pid%p.log -Dfile.encoding=UTF-8 -Duser.timezone=Asia/Shanghai -Dspring.profiles.active=huayang -jar <服务名>.jar > <服务名>.log 2>&1 &

# 6. 实时查看启动日志
tail -f <服务名>.log

# 7. 检查启动是否成功
grep "Started.*Application" <服务名>.log

# 8. 检查端口监听
netstat -tlnp | grep <端口号>

    日志查看与排查¶

    文件日志查看¶

    # 查看日志目录
ls -lh <日志目录>

# 实时追踪日志文件
tail -f <服务名>.log

# 查看最新 100 行
tail -n 100 <服务名>.log

# 搜索关键字
grep "关键词" <服务名>.log

# 搜索并显示前后 5 行
grep -C 5 "关键词" <服务名>.log

# 按时间过滤（今天的日志）
cat $(date +%Y-%m-%d).log

    日志排查命令¶

    # 查看最近的 ERROR 日志
grep "ERROR" <服务名>.log | tail -n 20

# 查看异常堆栈
tail -n 500 <服务名>.log | grep -E "ERROR|Exception|Failed|Caused by" | tail -n 30

# 搜索特定关键字
grep "关键词" <服务名>.log

# 查看启动成功标志
grep "Started.*Application" <服务名>.log

# 查看服务关闭信息
grep -i "shutdown\|stop\|close" <服务名>.log | tail -n 20

    Kafka 连接问题排查¶

    # 测试 Kafka 服务器连通性
telnet <kafka-ip> <kafka-port>

# 或使用 nc 命令
nc -zv <kafka-ip> <kafka-port>

# 查看 Kafka 消费者警告日志
grep "Connection to node" <服务名>.log

    时区与时间管理¶

    查看系统时间¶

    # 查看当前系统时间
date

# 查看当前时区
timedatectl

# 或者查看时区文件
cat /etc/timezone

# 查看系统时间戳
date +%s

    修改时区¶

    # 设置时区为 Asia/Shanghai
sudo timedatectl set-timezone Asia/Shanghai

# 或者使用旧命令
sudo ln -sf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime

# 同步系统时间（可选）
sudo ntpdate ntp.aliyun.com

    系统资源监控¶

    内存和 CPU¶

    # 查看内存使用情况
free -h

# 查看 CPU 使用情况
top
# 或
htop

# 查看磁盘使用情况
df -h

# 查看目录大小
du -sh <目录路径>

    网络连接¶

    # 查看网络连接
netstat -tulnp

# 查看 TCP 连接状态
netstat -an | grep -E "LISTEN|ESTABLISHED"

# 查看特定端口的连接数
netstat -an | grep <端口号> | wc -l

    常用场景速查¶

    场景 1：服务异常，需要重启¶

    # 1. 查找进程
ps aux | grep <服务名> | grep -v grep

# 2. 停止进程
kill -9 <PID>

# 3. 启动服务
nohup java -server -Xms1024m -Xmx2048m -XX:+HeapDumpOn -XX:HeapDumpPath=./heapdump -XX:ErrorFile=./hs_err_pid%p.log -Dfile.encoding=UTF-8 -Duser.timezone=Asia/Shanghai -Dspring.profiles.active=huayang -jar <服务名>.jar > <服务名>.log 2>&1 &

# 4. 查看日志
tail -f <服务名>.log

    场景 2：查看服务是否正常运行¶

    # 查看进程
ps aux | grep <服务名> | grep -v grep

# 查看端口
netstat -tlnp | grep <端口号>

# 查看日志
tail -n 50 <服务名>.log

    场景 3：实时追踪日志排查问题¶

    tail -f <服务名>.log | grep "关键词"

    场景 4：排查启动失败¶

    # 查看完整日志
tail -n 200 <服务名>.log

# 查找错误信息
grep -i "error\|exception\|failed" <服务名>.log

# 查看配置文件是否存在
ls -la application*.yml application*.properties

    场景 5：服务进程消失排查¶

    排查步骤

    服务进程消失通常是由于 OOM Killer 或系统资源不足导致的。

    # 查看系统日志
tail -n 100 /var/log/messages
# 或
tail -n 100 /var/log/syslog

# 查看 OOM Killer 日志
dmesg | grep -i "out of memory"
dmesg | grep -i "killed process"

# 查看 nohup.out
cat nohup.out

    实用小技巧¶

    快速查找进程和端口¶

    # 一键查看进程和端口
ps aux | grep <服务名> | grep -v grep && netstat -tlnp | grep <进程名>

    批量清理日志文件¶

    # 清理 7 天前的日志
find . -name "*.log" -mtime +7 -delete

# 或压缩旧日志
find . -name "*.log" -mtime +7 -exec gzip {} \;

    监控服务状态脚本¶

    #!/bin/bash
SERVICE="<服务名>"
PID=$(ps aux | grep $SERVICE | grep -v grep | awk '{print $2}')

if [ -z "$PID" ]; then
    echo "服务未运行，正在启动..."
    nohup java -server -Xms1024m -Xmx2048m -jar ${SERVICE}.jar > ${SERVICE}.log 2>&1 &
    echo "服务已启动"
else
    echo "服务正在运行，PID: $PID"
fi
