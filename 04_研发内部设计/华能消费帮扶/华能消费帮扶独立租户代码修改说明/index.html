    华能消费帮扶独立租户代码修改说明¶

    文档概述

    本文档详细说明华能消费帮扶独立租户功能的代码实现，包括新增文件、修改文件、代码修复计划等内容。

    概述¶

    核心功能¶

    本次编码实现了华能消费帮扶商品的独立租户路由功能，通过配置化的方式将消费帮扶商品（goodsType="99"）从华能租户（1706958926295019522）自动路由到华能消费帮扶租户（2018950843406299138），实现数据完全隔离。

    核心思路: 采用租户→商品类型映射关系表，动态获取目标租户ID，避免硬编码，提高系统扩展性。

    核心代码: TenantGoodsTypeMappingService 处理租户分配业务，基于新增的租户→商品类型映射关系表获取对应租户 id。为了避免模块依赖问题，当前代码分别在需要的模块中创建了独立的服务实现，涉及服务模块：bssc-biz-operate、bssc-biz-interchange。

    数据源编码: - 原数据源拼接后缀 -xfbf（测试账号数据源：hncs-xfbf-zncszh）

    AI端数据源编码: hnzhgyl-xfbf

    租户id: - 测试环境使用租户：1650033261075759106 进行测试 - 新增测试华能帮扶租户 -- 2019308420509958145
    - 1706958926295019522 -- 生产华能租户id - 生产消费帮扶租户id -- 2018950843406299138

    爬虫监测源: 待定--832平台，央企消费帮扶平台，京东自营

@Data
@EqualsAndHashCode()
@NoArgsConstructor
@AllArgsConstructor
@TableName("t_tenant_goods_type_mapping")
public class TenantGoodsTypeMappingEntity {

    /**
     * 主键ID
     */
    @TableId(type = IdType.AUTO)
    private Long id;

    /**
     * 原始租户ID
     */
    @TableField("original_tenant_id")
    private String originalTenantId;

    /**
     * 目标租户ID
     */
    @TableField("target_tenant_id")
    private String targetTenantId;

    /**
     * 商品类型：99=消费帮扶商品
     */
    @TableField("goods_type")
    private String goodsType;

    /**
     * 商品类型名称：99=消费帮扶商品
     */
    @TableField("goods_type_name")
    private String goodsTypeName;

    /**
     * 映射描述
     */
    @TableField("description")
    private String description;

    /**
     * 状态：1=启用，0=禁用
     */
    @TableField("status")
    private Integer status;

    /**
     * 创建时间
     */
    @TableField(value = "create_time", fill = FieldFill.INSERT)
    private LocalDateTime createTime;

    /**
     * 更新时间
     */
    @TableField(value = "update_time", fill = FieldFill.INSERT_UPDATE)
    private LocalDateTime updateTime;

    /**
     * 创建人
     */
    @TableField("create_user")
    private String createUser;

    /**
     * 修改人
     */
    @TableField("update_user")
    private String updateUser;

    /**
     * 是否删除：0=否，1=是
     */
    @TableField("is_delete")
    private Integer isDelete;

    /**
     * 逻辑删除标志
     */
    @TableLogic
    private Integer deleted;

    /**
     * 商品类型枚举
     */
    public enum GoodsType {
        CONSUMPTION_ASSISTANCE("99", "消费帮扶商品");

        private String value;
        private String name;

        GoodsType(String value, String name) {
            this.value = value;
            this.name = name;
        }

        public String getValue() {
            return this.value;
        }

        public String getName() {
            return this.name;
        }

        public static String getGoodsTypeName(String goodsType) {
            for (GoodsType type : GoodsType.values()) {
                if (type.value.equals(goodsType)) {
                    return type.name;
                }
            }
            return "";
        }
    }
}

import com.bssc.maint.operate.db.mapper.TenantGoodsTypeMappingMapper;
import com.bssc.maint.operate.db.model.TenantGoodsTypeMappingEntity;
import com.bssc.maint.operate.service.TenantGoodsTypeMappingService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.cache.annotation.CacheEvict;
import org.springframework.cache.annotation.Cacheable;
import org.springframework.stereotype.Service;

import javax.annotation.Resource;
import java.time.LocalDateTime;
import java.util.List;

/**
 * 租户商品类型映射服务实现类
 * 
 * @author lilinlin
 * @since 2026-02-03
 */
@Slf4j
@Service
public class TenantGoodsTypeMappingServiceImpl implements TenantGoodsTypeMappingService {

    @Resource
    private TenantGoodsTypeMappingMapper mappingMapper;

    @Override
    public String getTargetTenantId(String originalTenantId, String goodsType) {
        if (originalTenantId == null || goodsType == null) {
            log.info("租户ID或商品类型为空，返回原始租户ID");
            return originalTenantId;
        }

        TenantGoodsTypeMappingEntity mapping = mappingMapper.selectTargetTenant(originalTenantId, goodsType);

        if (mapping != null) {
            log.info("租户映射：{} + {} → {}", originalTenantId, goodsType, mapping.getTargetTenantId());
            return mapping.getTargetTenantId();
        }

        // 没有映射规则则返回原租户ID
        log.info("租户 {} 的商品类型 {} 无映射规则，使用原始租户", originalTenantId, goodsType);
        return originalTenantId;
    }

    @Override
    public boolean isHuaNengTenantId(String tenantId) {
        // 是否华能租户

    @Override
    public void setTenantCatalogsCache(RedisTemplate redisTemplate, String tenantId, List<? extends Object> tenantCatalogsList) {
        if (redisTemplate == null || tenantId == null || tenantCatalogsList == null) {
            log.error("设置租户品目缓存参数为空");
            return;
        }
        // lilinlin 华能消费帮扶需求，设置源租户缓存
        String key = CacheKey.TENANT_CATALOGS + tenantId;
        redisTemplate.opsForList().leftPushAll(key, tenantCatalogsList);
        log.info("设置租户 {} 品目缓存成功", tenantId);

        // lilinlin 华能消费帮扶需求，为虚拟租户设置相同缓存
        if (isHuaNengTenantId(tenantId)) {
            List<TenantGoodsTypeMappingEntity> mappings = getMappingsByOriginalTenantId(tenantId);
            if (mappings != null && !mappings.isEmpty()) {
                for (TenantGoodsTypeMappingEntity mapping : mappings) {
                    String targetTenantId = mapping.getTargetTenantId();
                    String targetKey = CacheKey.TENANT_CATALOGS + targetTenantId;
                    redisTemplate.opsForList().leftPushAll(targetKey, tenantCatalogsList);
                    log.info("设置虚拟租户 {} 品目缓存成功（来源租户 {}）", targetTenantId, tenantId);
                }
            }
        }
    }

    @Override
    public void deleteTenantCatalogsCache(RedisTemplate redisTemplate, String tenantId) {
        if (redisTemplate == null || tenantId == null) {
            log.error("删除租户品目缓存参数为空");
            return;
        }
        // lilinlin 华能消费帮扶需求，删除源租户缓存
        String key = CacheKey.TENANT_CATALOGS + tenantId;
        redisTemplate.delete(key);
        log.info("删除租户 {} 品目缓存成功", tenantId);

        // lilinlin 华能消费帮扶需求，删除虚拟租户缓存
        if (isHuaNengTenantId(tenantId)) {
            List<TenantGoodsTypeMappingEntity> mappings = getMappingsByOriginalTenantId(tenantId);
            if (mappings != null && !mappings.isEmpty()) {
                for (TenantGoodsTypeMappingEntity mapping : mappings) {
                    String targetTenantId = mapping.getTargetTenantId();
                    String targetKey = CacheKey.TENANT_CATALOGS + targetTenantId;
                    redisTemplate.delete(targetKey);
                    log.info("删除虚拟租户 {} 品目缓存成功（来源租户 {}）", targetTenantId, tenantId);
                }
            }
        }
    }
